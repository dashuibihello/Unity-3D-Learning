# 2.请用三种方法以上方法，实现物体的抛物线运动。（如，修改Transform属性，使用向量Vector3的方法…）
## 方法一：
	public class BallBeh1 : MonoBehaviour {
    public float y = 1;
    public float time = 0;
    // Use this for initialization
    void Start () {

	}
	
	// Update is called once per frame
	void Update () {
        this.transform.position += Vector3.right * Time.deltaTime;
        this.transform.position += Vector3.down * (Time.deltaTime * y / 100);
        time++;
        y = 9.8F * time;
    }
	}
## 方法二：
	public class BallBeh2 : MonoBehaviour {
    public float y = 0;
    public float time = 0;
    // Use this for initialization
    void Start () {
        Debug.Log("Ball Start");
    }
	
	// Update is called once per frame
	void Update () {
        transform.Translate(Time.deltaTime, -Time.deltaTime * y / 100, 0);
        time++;
        y = 9.8F * time;
    }
	}
## 方法三：
	public class BallBeh3 : MonoBehaviour {
    Vector3 target;
    public float x = 0;
    public float y = 0;
    public float time = 0;
    // Use this for initialization

    void Start() {
        
    }

    void Update() {
        target = new Vector3(x, -y, 0);
        transform.position = Vector3.MoveTowards(transform.position, target, 5 * Time.deltaTime);
        time++;
        x += 100;
        y = 9.8F / 2 * time * time;
    }
	}
